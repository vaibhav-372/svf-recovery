svf_bank_recovery_app:
  description: A mobile application for managing bank recovery operations with features like location tracking, admin dashboard, and agent performance tracking.
  technologies:
    - React Native
    - Node.js
    - MongoDB
    - Express.js
    - Redux
    - Axios
  features:
    - Location Tracking:
        description: Track the real-time location of agents during recovery operations.
        implementation: Use React Native's Geolocation API and integrate with a mapping service (e.g., Google Maps).
    - Admin Dashboard:
        description: A web-based dashboard for administrators to monitor operations, view reports, and manage agents.
        implementation: Build a React web application with a RESTful API backend using Node.js and Express.js.
    - User Authentication:
        description: Secure user authentication for agents and administrators.
        implementation: Use JWT (JSON Web Tokens) for authentication and authorization.
    - Image Upload with Location Coordinates:
        description: Allow agents to upload images related to recovery operations along with their current location coordinates.
        implementation: File upload with library (Multer) in Node.js and store images in a cloud storage service (AWS S3)
    - Agent Performance Tracking:
        description: Track the performance of agents based on completed tasks, and customer feedback.
        implementation: Create a performance metrics system using MongoDB to store agent data and generate reports.

  database_schema:
    users:
      fields:
        - userId: String (unique identifier)
        - name: String 
        - role: String (admin/agent)
        - email: String
        - password: String (hashed)
    locations:
      fields:
        - locationId: String (unique identifier)
        - userId: String (reference to users)
        - latitude: Number
        - longitude: Number
        - timestamp: Date
    images:
      fields:
        - imageId: String (unique identifier)
        - userId: String (reference to users)
        - imageUrl: String (URL of the uploaded image)
        - coordinates: {
            latitude: Number,
            longitude: Number
          }
        - timestamp: Date
    performance_metrics:
      fields:
        - metricId: String (unique identifier)
        - userId: String (reference to users)
        - tasksCompleted: Number
        - responseTimeAvg: Number
        - feedbackScoreAvg: Number
    api_endpoints:
      - GET /api/users: Retrieve all users
      - POST /api/users: Create a new user
      - GET /api/locations: Retrieve all locations
      - POST /api/locations: Add a new location
      - GET /api/images: Retrieve all images
      - POST /api/images: Upload a new image
      - GET /api/performance: Retrieve performance metrics for agents
      - POST /api/performance: Update performance metrics for an agent
