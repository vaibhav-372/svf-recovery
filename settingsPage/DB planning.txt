DB planning for tbl bank recovery application with react native and node.js (location tracking, admin dashboard, image upload with location coordinates, agent performance tracking, and more)

tbl_bank_recovery_app:
  technologies:
    - React Native
    - Node.js 
    - Express.js 
    - Redux
    - Axios
    - MYSQL


tbl_emp_profile
    -entry_id                   AI
    -emp_id                     int
    -emp_name                   varchar(100)
    -emp_role                   varchar(5)('admin', 'agent', 'operator')
    -mobile1                    varchar(30)
    -mobile2                    varchar(30)
    -email                      varchar(100)
    -password                   varchar(100)
    -joined_date                Date
    -last_login                 Date
    -status                     varchar(10)('active', 'inactive')
    -visited_locations          int
    -worked_days                int
    -failed_tasks               int
    -completed_tasks            int
    -average_response_time      float   (optional)
    -profile_image_url          varchar(255)
    -created_date               DateTime 
    -last_updated_date          DateTime    
    -isBlocked                  boolean (indicates if the emp is blocked)
    -isDeleted                  boolean (indicates if the emp is deleted)
    -device_id                  varchar(50)
    -created_by                 int
    -branch_id                  int
    -reffered_by                varchar(100)


tbl_location_tracking       ---(optional)
    -entry_id                   AI
    -emp_id                     int (foreign key to tbl_emp_profile.emp_id)
    -latitude                   float 
    -longitude                  float
    -accuracy                   float ( To track the accuracy of the location in meters)
    -created_time               Time
    -device_id                  varchar(100)


tbl_auction_customers
    -entry_id                   AI
    -pt_no                      int (unique identifier for the customer)
    -customer_id                int
    -customer_name              varchar(100)
    -customer_photo_url         varchar(255) (optional, if an image is uploaded)
    -city                       varchar(100)  (updated from area)
    -address                    varchar(255)
    -contact_number1            int(10)
    -contact_number2            int(10)
    -nominee_name               varchar(100)
    -nominee_contact_number     varchar(30) (optional)
    -ornament_name              varchar(100)
    -loan_created_date          DateTime
    -interest_rate              float 
    -gross_weight               float (in grams)
    -net_weight                 float (in grams)    
    -tenure                     int
    -loan_amount                float
    -paid_amount                float (amount already paid by the customer)
    -total_amount               float (total amount to be paid by the customer)
    -first_letter_date          Date
    -second_letter_date         Date
    -final_letter_date          Date (optional, if a final letter is sent)
    -assigned_date              DateTime
    -last_date                  DateTime (last date to pay loan amount)
    -created_by                 int (foreign key to tbl_emp_profile.emp_id)
    -branch_id                  int
    -device_id                  varchar(100)


tbl_assigned_agents
    -entry_id                    AI
    -cus_auction_id              int (foreign key to tbl_auction_customers.entry_id)
    -pt_no                       int (foreign key to tbl_auction_customers.pt_no)
    -assigned_emp_id             int (foreign key to tbl_profile.emp_id)
    -agent_id                    int (foreign key to tbl_emp_profile.emp_id)
    -assigned_date               DateTime 
    -isVisited                   boolean (indicates if the customer has been visited by an agent)
    -identifier                  int (to track the unique identifier for the assignment)
    -device_id                   varchar(100) (optional, to track the device used for assignment)
    -created_by                  int (foreign key to tbl_emp_profile.emp_id)    
    -branch_id                   int


tbl_recovery_responses
    -entry_id                   AI
    -assigned_agent_id          int (foreign key to tbl_assigned_agents.entry_id)
    -cus_auction_id             int (foreign key to tbl_auction_customers.entry_id)
    -pt_no                      int (unique identifier for the response)
    -customer_id                int (foreign key to tbl_recovery_customers.customer_id)
    -agent_id                   int (foreign key to tbl_profile.emp_id)
    -response_text              text
    -response_status            int (0 for fail, 1 for success)
    -response_timestamp         DateTime
    -interest_amount            float (optional, if the response includes interest amount)
    -image_url                  varchar(255) (optional, if an image is uploaded with the response)
    -location_coordinates       json ({"latitude": float, "longitude": float}) (optional)
    -completed_date             DateTime (optional, if the response is completed
    -device_id                  varchar(100) (optional, to track the device used for the response)
    -branch_id                  int
    -last_updated_at            DateTime

tbl_recovery_notifications      -(optional)
    -entry_id                   AI
    -emp_id                     int (foreign key to tbl_emp_profile.emp_id)
    -notification_type          varchar(50)
    -message                    text
    -created_at                 DateTime
    -device_id                  varchar(100) (optional, to track the device used for the notification)
    -branch_id                  int 


    api_endpoints:
    admin:
      - GET         admin/api/getAgents: Retrieve all agents
      - POST        admin/api/createAgent: Create a new agent
      - PUT         admin/api/updateAgent/:id: Update a specific agent by ID
      - DELETE      admin/api/deleteAgent/:id: Delete a specific agent by ID
      - GET         admin/api/getCustomers: Retrieve all customers
      - GET         agent/api/customers/ Retrieve all customers assigned to the agent
      - GET         agent/api/customers/:id: Retrieve a specific customers by agent ID
      - GET         agent/api/responses/ Retrieve a all responses
      - GET         agent/api/responses/:id: Retrieve a specific response by agent ID
      - POST        admin/api/createNotification: Create a new notification
      - GET         admin/api/getNotifications: Retrieve all notifications

    agent:
      - GET         agent/api/customers/ Retrieve all customers assigned to the agent
      - GET         agent/api/customers/:id: Retrieve a specific customer by ID
      - POST        agent/api/responses: Create a new response for a customer
      - GET         agent/api/responses/:id: Retrieve a specific response by ID 
      - PUT         agent/api/responses/:id: Update a specific response by ID
      - GET         agent/api/getNotifications/:id: Retrieve notifications for a specific agent



* same PTno do not exist and can't be assigned to multiple agents
